# 우리농 자가진단 체크리스트 — AI 분석 엔진 전환 작업지시서

> ⚠ 이 문서는 기존 규칙 기반 엔진(checklist_engine.py 등)을 
> **Claude API 기반 AI 분석 엔진**으로 전면 교체하는 작업입니다.

---

## 왜 전환하는가

기존 규칙 기반(if-else + 키워드 매칭) 엔진은:
- 물품사양서의 맥락적 판단 불가 (원재료 분류 경계 사례 등)
- 증빙서류의 내용 적합성 판단 불가 (제품명 대조, 유효기간 해석 등)
- 신입직원에게 유용한 수준의 설명 생성 불가

Claude API를 호출하면 Claude Projects에서와 동일한 분석 품질을 웹앱에서 제공할 수 있다.

---

## 아키텍처

```
사용자 (브라우저)
    │
    ▼
Flask 서버 (Render)
    │
    ├─ PDF/이미지 업로드 수신
    ├─ PDF → 텍스트 추출 (pdfplumber, 기존 코드 유지)
    ├─ 이미지 → base64 인코딩
    ├─ 파일 해시 계산 → 캐시 확인
    │
    ├─ [캐시 HIT] → 저장된 분석 결과 즉시 반환
    │
    ├─ [캐시 MISS] → Claude API 호출
    │   ├─ system: 시스템 프롬프트 (분석 절차 + 출력 형식)
    │   ├─ cache_control: 규정 지식베이스 (prompt caching)
    │   ├─ user: 추출된 PDF 텍스트 + 이미지
    │   └─ → AI 분석 결과 수신
    │
    ├─ 결과를 캐시에 저장
    ├─ 결과를 HTML로 렌더링
    └─ 결과를 엑셀/PDF로 다운로드 제공
```

### 1. engine/ai_analyzer.py (신규)

이 파일이 핵심입니다. 기존의 checklist_engine.py, doc_matcher.py, label_comparator.py를 
**하나의 AI 호출**로 대체합니다.

```python
import os
import json
import hashlib
import urllib.request
from pathlib import Path

API_KEY = os.environ.get("ANTHROPIC_API_KEY")
API_URL = "https://api.anthropic.com/v1/messages"
MODEL = "claude-sonnet-4-20250514"
MAX_TOKENS = 8000

# 캐시 디렉토리
CACHE_DIR = Path("cache")
CACHE_DIR.mkdir(exist_ok=True)

def get_cache_key(*contents):
    """업로드된 파일 내용들의 해시로 캐시 키 생성"""
    combined = "".join(str(c) for c in contents)
    return hashlib.sha256(combined.encode()).hexdigest()[:16]

def get_cached_result(cache_key):
    """캐시된 결과가 있으면 반환"""
    cache_file = CACHE_DIR / f"{cache_key}.json"
    if cache_file.exists():
        return json.loads(cache_file.read_text(encoding="utf-8"))
    return None

def save_to_cache(cache_key, result):
    """분석 결과를 캐시에 저장"""
    cache_file = CACHE_DIR / f"{cache_key}.json"
    cache_file.write_text(json.dumps(result, ensure_ascii=False, indent=2), encoding="utf-8")

def load_knowledge_base():
    """규정 지식베이스 로드"""
    kb_path = Path("data/01_규정_지식베이스_v2.md")
    if kb_path.exists():
        return kb_path.read_text(encoding="utf-8")
    # 폴백: data 폴더에 없으면 docs에서
    kb_path = Path("docs/01_규정_지식베이스_v2.md")
    if kb_path.exists():
        return kb_path.read_text(encoding="utf-8")
    raise FileNotFoundError("규정 지식베이스 파일을 찾을 수 없습니다")

def load_system_prompt():
    """시스템 프롬프트 로드 — 웹앱용으로 수정된 버전"""
    return SYSTEM_PROMPT_WEBAPP

def call_claude_api(system_prompt, knowledge_base, user_message, images=None):
    """Claude API 호출 (prompt caching 적용)"""
    if not API_KEY:
        return {"error": "ANTHROPIC_API_KEY가 설정되지 않았습니다. 포장지 사진 분석과 AI 분석 기능을 사용하려면 API 키가 필요합니다."}
    
    # 메시지 구성
    system_messages = [
        {
            "type": "text",
            "text": system_prompt
        },
        {
            "type": "text",
            "text": knowledge_base,
            "cache_control": {"type": "ephemeral"}  # prompt caching
        }
    ]
    
    # user 메시지 구성
    user_content = []
    
    # 이미지가 있으면 추가 (포장지 사진)
    if images:
        for img in images:
            user_content.append({
                "type": "image",
                "source": {
                    "type": "base64",
                    "media_type": img["media_type"],
                    "data": img["data"]
                }
            })
    
    user_content.append({
        "type": "text",
        "text": user_message
    })
    
    payload = {
        "model": MODEL,
        "max_tokens": MAX_TOKENS,
        "system": system_messages,
        "messages": [
            {"role": "user", "content": user_content}
        ]
    }
    
    headers = {
        "Content-Type": "application/json",
        "x-api-key": API_KEY,
        "anthropic-version": "2023-06-01",
        "anthropic-beta": "prompt-caching-2024-07-31"
    }
    
    req = urllib.request.Request(
        API_URL,
        data=json.dumps(payload).encode("utf-8"),
        headers=headers,
        method="POST"
    )
    
    try:
        with urllib.request.urlopen(req, timeout=120) as resp:
            result = json.loads(resp.read().decode("utf-8"))
            # 텍스트 응답 추출
            text_parts = [block["text"] for block in result["content"] if block["type"] == "text"]
            return {
                "analysis": "\n".join(text_parts),
                "usage": result.get("usage", {}),
                "cached": False
            }
    except urllib.error.HTTPError as e:
        error_body = e.read().decode("utf-8") if e.readable() else str(e)
        return {"error": f"API 호출 실패 ({e.code}): {error_body}"}
    except Exception as e:
        return {"error": f"API 호출 중 오류: {str(e)}"}


def analyze_quick_diagnosis(pdf_text=None, images=None):
    """기능 A: 빠른 진단"""
    knowledge_base = load_knowledge_base()
    system_prompt = load_system_prompt()
    
    # 캐시 확인
    cache_key = get_cache_key(pdf_text or "", str(images is not None))
    cached = get_cached_result(cache_key)
    if cached:
        cached["cached"] = True
        return cached
    
    # 사용자 메시지 구성
    if pdf_text and images:
        user_msg = f"""다음은 신규 물품의 포장지 사진과 물품사양서입니다. 
기능 A(빠른 진단)를 수행해주세요.

[물품사양서 내용]
{pdf_text}

포장지 사진도 함께 첨부되어 있습니다. 포장지의 표시사항도 함께 분석해주세요."""
    elif pdf_text:
        user_msg = f"""다음은 신규 물품의 물품사양서입니다.
기능 A(빠른 진단)를 수행해주세요.

[물품사양서 내용]
{pdf_text}"""
    elif images:
        user_msg = """다음은 신규 물품의 포장지 사진입니다.
기능 A(빠른 진단)를 수행해주세요.
포장지의 표시사항을 읽고 분석해주세요."""
    else:
        return {"error": "분석할 파일이 없습니다."}
    
    result = call_claude_api(system_prompt, knowledge_base, user_msg, images)
    
    if "error" not in result:
        save_to_cache(cache_key, result)
    
    return result


def analyze_document_verification(pdf_text, supporting_docs_text, images=None):
    """기능 B: 서류 검증"""
    knowledge_base = load_knowledge_base()
    system_prompt = load_system_prompt()
    
    # 캐시 확인
    cache_key = get_cache_key(pdf_text, supporting_docs_text, str(images is not None))
    cached = get_cached_result(cache_key)
    if cached:
        cached["cached"] = True
        return cached
    
    # 사용자 메시지 구성
    user_msg = f"""다음은 신규 물품의 물품사양서와 증빙서류입니다.
기능 B(서류 검증)를 수행해주세요.

[물품사양서 내용]
{pdf_text}

[증빙서류 내용]
{supporting_docs_text}"""

    if images:
        user_msg += "\n\n포장지 사진도 함께 첨부되어 있습니다. 포장지의 표시사항과 물품사양서의 내용 일치 여부도 검증해주세요."
    
    result = call_claude_api(system_prompt, knowledge_base, user_msg, images)
    
    if "error" not in result:
        save_to_cache(cache_key, result)
    
    return result

당신은 가톨릭농민회/우리농촌살리기운동본부의 **신규물품 출하 준비 자가진단 도우미**입니다.
지역 교구 우리농 생산소비 담당 실무자가 사용하며, 신입직원도 이해할 수 있도록 친절하게 설명합니다.

## 두 가지 기능

### 기능 A: 빠른 진단
포장지 사진이나 물품사양서를 받으면, 출하에 필요한 서류와 주의사항을 안내합니다.

출력 형식 (반드시 이 구조를 따르세요):

#### 📦 제품 기본정보
제품명, 식품유형, 교구, 생산자, 회원구분, 제조원 등을 정리합니다.

#### 📋 출하에 필요한 서류 체크리스트
각 서류에 대해:
- 서류명
- 왜 필요한지 (신입직원도 이해할 수 있는 설명)
- 이 제품에서 특별히 주의할 점

필수 서류와 이 제품의 특성에 따른 조건부 서류를 구분하여 표시합니다.

#### ⚠️ 주의사항

- 무엇이 문제인지

논의가 필요하거나 이슈 가능성이 있다면 반드시 포함합니다.

원재료가 원부재료/첨가물/향신료 중 어디에 해당하는지 애매한 경우 
(예: 바나나농축분말처럼 농산물 가공품이면서 제품명에 포함된 경우),
각 분류별 적용 기준 차이를 설명하고 위원회 논의를 권고합니다.

#### 📨 생산자에게 요청할 사항

### 기능 B: 서류 검증
물품사양서 + 증빙서류(+ 선택적으로 포장지 사진)를 받으면 검증합니다.

출력 형식 (반드시 이 구조를 따르세요):

#### 1️⃣ 포장지 ↔ 물품사양서 대조 (포장지 사진이 있는 경우만)
항목별로 일치/불일치를 테이블로 정리합니다.

#### 2️⃣ 물품사양서 기재 점검
20개 필수 항목의 기재 여부를 점검합니다.
미기재/오류 항목에 대해 무엇을 어떻게 보완해야 하는지 설명합니다.

#### 3️⃣ 증빙서류 점검
각 필요 서류의 제출/미제출 여부를 표시합니다.
제출된 서류에 대해:
- 제품명이 물품사양서와 일치하는지
- 유효기간이 남아있는지
- 검사 결과가 적합한지
문제가 발견되면 구체적으로 설명합니다.

#### 4️⃣ 출하 승인 시 예상 이슈
전국물품위원회 심의에서 논의될 수 있는 사항을 미리 알려줍니다.
실무자가 사전에 대비할 수 있도록 합니다.

#### 📨 추가 요청이 필요한 서류
미제출 서류와 보완 사항을 생산자에게 전달할 수 있는 형태로 정리합니다.

## 분석 원칙

1. **규정 근거주의**: 판단의 근거를 규정 조항으로 명시합니다.
2. **보수적 판단**: 확실하지 않으면 "확인 필요"로 표시합니다.
3. **친절한 설명**: 모든 항목에 신입직원도 이해할 수 있는 설명을 달아줍니다.
4. **놓치지 않기**: 규정에 명시적 근거가 없는 이슈라도, 출하 승인 과정에서 
   문제될 가능성이 조금이라도 있으면 주의사항에 포함합니다.
5. **실무 지향**: "어떻게 해결하면 되는지"를 반드시 함께 안내합니다.

- 테이블이 필요한 곳에는 마크다운 테이블을 사용합니다
```

### 2. app.py 수정

기존 라우트를 수정하여 AI 분석 엔진을 호출하도록 변경합니다.

```python
# 기존 checklist_engine 임포트를 ai_analyzer로 교체
from engine.ai_analyzer import analyze_quick_diagnosis, analyze_document_verification

# PDF 텍스트 추출은 기존 pdf_parser.py 유지
from engine.pdf_parser import extract_text_from_pdf

import base64

@app.route('/api/analyze-quick', methods=['POST'])
def analyze_quick():
    """기능 A: 빠른 진단 — AI 분석"""
    pdf_text = None
    images = None
    
    # PDF 처리
    if 'pdf_file' in request.files:
        pdf_file = request.files['pdf_file']
        if pdf_file.filename:
            pdf_path = save_upload(pdf_file)
            pdf_text = extract_text_from_pdf(pdf_path)
    
    # 이미지 처리 (포장지 사진)
    if 'label_image' in request.files:
        img_file = request.files['label_image']
        if img_file.filename:
            img_data = base64.b64encode(img_file.read()).decode('utf-8')
            media_type = img_file.content_type or 'image/jpeg'
            images = [{"data": img_data, "media_type": media_type}]
    
    result = analyze_quick_diagnosis(pdf_text=pdf_text, images=images)
    return jsonify(result)


@app.route('/api/analyze-verify', methods=['POST'])
def analyze_verify():
    """기능 B: 서류 검증 — AI 분석"""
    pdf_text = None
    supporting_docs_text = ""
    images = None
    
    # 물품사양서 PDF
    if 'pdf_file' in request.files:
        pdf_file = request.files['pdf_file']
        if pdf_file.filename:
            pdf_path = save_upload(pdf_file)
            pdf_text = extract_text_from_pdf(pdf_path)
    
    # 증빙서류 (복수)
    if 'supporting_docs' in request.files:
        docs = request.files.getlist('supporting_docs')
        doc_texts = []
        for i, doc in enumerate(docs):
            if doc.filename:
                doc_path = save_upload(doc)
                if doc.filename.lower().endswith('.pdf'):
                    text = extract_text_from_pdf(doc_path)
                    doc_texts.append(f"--- 증빙서류 {i+1}: {doc.filename} ---\n{text}")
                else:
                    # 이미지 증빙서류는 별도 처리 가능
                    doc_texts.append(f"--- 증빙서류 {i+1}: {doc.filename} (이미지 파일) ---")
        supporting_docs_text = "\n\n".join(doc_texts)
    
    # 포장지 사진
    if 'label_image' in request.files:
        img_file = request.files['label_image']
        if img_file.filename:
            img_data = base64.b64encode(img_file.read()).decode('utf-8')
            media_type = img_file.content_type or 'image/jpeg'
            images = [{"data": img_data, "media_type": media_type}]
    
    if not pdf_text:
        return jsonify({"error": "물품사양서 PDF를 업로드해주세요."})
    
    result = analyze_document_verification(
        pdf_text=pdf_text,
        supporting_docs_text=supporting_docs_text,
        images=images
    )
    return jsonify(result)
```

```html
// 분석 결과를 마크다운 → HTML로 렌더링
function displayResult(data) {
  const resultArea = document.getElementById('result-area');
  const resultContent = document.getElementById('result-content');
  
  if (data.error) {
    resultContent.innerHTML = `<div class="error-box">${data.error}</div>`;

        + resultContent.innerHTML;
    }
  }
  
  resultArea.style.display = 'block';

  const resultContent = document.getElementById('result-content');
  const headings = resultContent.querySelectorAll('h4');
  let requestSection = '';
  
  for (const h of headings) {
    if (h.textContent.includes('요청') || h.textContent.includes('📨')) {
      let node = h.nextElementSibling;
      while (node && node.tagName !== 'H4') {
        requestSection += node.textContent + '\n';
        node = node.nextElementSibling;
      }
      break;
    }
  }
  
  if (requestSection) {
    navigator.clipboard.writeText(requestSection.trim());
    alert('요청사항이 복사되었습니다.');
  } else {
    // 전체 결과 복사
    navigator.clipboard.writeText(resultContent.textContent);
    alert('분석 결과가 복사되었습니다.');
  }
}
```

### 기능 1: "새로 분석하기" 버튼

캐시된 결과가 표시될 때 (💾 이전 분석 결과 안내와 함께), 
**"🔄 새로 분석하기"** 버튼을 표시합니다.

이 버튼을 누르면:
- 캐시를 무시하고 API를 다시 호출
- 서버 측: API 엔드포인트에 `?force=true` 파라미터 추가
- ai_analyzer.py에서 force=True이면 캐시 조회를 건너뛰고 바로 API 호출
- 새 결과로 캐시를 덮어씀

```javascript
let lastAnalysisContext = null;

async function reanalyze() {
  if (!lastAnalysisContext) return;
  const { endpoint, formData } = lastAnalysisContext;
  showLoading();
  try {
    // force=true 파라미터 추가
    const url = endpoint + (endpoint.includes('?') ? '&' : '?') + 'force=true';
```

분석 결과가 표시된 상태에서, 다른 물품을 분석하고 싶을 때

- 파일 input 값 초기화
- 분석 결과 영역 숨김
- 페이지 상단으로 스크롤
- 현재 선택된 탭(빠른 진단 / 서류 검증)은 유지

```javascript
function resetAll() {
  // 1. 파일 input 초기화
  const fileInputs = document.querySelectorAll('input[type="file"]');
  fileInputs.forEach(input => { input.value = ''; });
  
  // 2. 파일 목록 표시 영역 초기화
  const fileLists = document.querySelectorAll('.file-list');
  fileLists.forEach(list => { list.innerHTML = ''; });
  
  // 3. 분석 결과 영역 숨김
  const resultArea = document.getElementById('result-area');
  if (resultArea) {
    resultArea.style.display = 'none';
    document.getElementById('result-content').innerHTML = '';
  }
  
  // 4. 캐시 알림 숨김
  const cacheNotice = document.getElementById('cache-notice');
  if (cacheNotice) cacheNotice.style.display = 'none';
  
  // 5. 마지막 분석 컨텍스트 초기화
  lastAnalysisContext = null;
  
  // 6. 페이지 상단으로 스크롤
  window.scrollTo({ top: 0, behavior: 'smooth' });
}
```
